Certainly! To create a RESTful API using Python Flask for your movie recommender with the specified requirements, you can follow the example code below. Please note that this is a basic example, and you may need to adapt it based on your specific requirements and the structure of your movie recommender system.

python

from flask import Flask, request, jsonify, make_response
from flask_restful import Api, Resource
import csv

app = Flask(__name__)
api = Api(app)

# In-memory user storage for demonstration purposes
users = {
    'username': 'password'
}

# In-memory token storage for demonstration purposes
tokens = {}

# Dummy movie data for demonstration purposes
movies = [
    {"id": 1, "title": "Movie 1", "genre": "Action"},
    {"id": 2, "title": "Movie 2", "genre": "Comedy"},
    # Add more movie data as needed
]

# Helper function to validate authentication token
def authenticate_user(token):
    return token in tokens

# Resource for movie recommendation
class MovieRecommendation(Resource):
    def get(self):
        # Check if user is authenticated
        token = request.headers.get('X-OBSERVATORY-AUTH')
        if not authenticate_user(token):
            return make_response(jsonify({'error': 'Not authorized'}), 401)

        # Get format parameter from the query string, default to 'json'
        data_format = request.args.get('format', 'json')

        if data_format == 'json':
            return jsonify(movies)
        elif data_format == 'csv':
            # Generate CSV response
            csv_data = "id,title,genre\n"
            for movie in movies:
                csv_data += f"{movie['id']},{movie['title']},{movie['genre']}\n"

            response = make_response(csv_data)
            response.headers['Content-Type'] = 'text/csv; charset=utf-8'
            return response
        else:
            return make_response(jsonify({'error': 'Invalid format parameter'}), 400)

# Resource for user login
class UserLogin(Resource):
    def post(self):
        username = request.form.get('username')
        password = request.form.get('password')

        if username in users and users[username] == password:
            # Generate and store token
            token = f"{username}_token"
            tokens[token] = username

            return jsonify({'token': token})
        else:
            return make_response(jsonify({'error': 'Bad request'}), 400)

# Resource for user logout
class UserLogout(Resource):
    def post(self):
        # Check if user is authenticated
        token = request.headers.get('X-OBSERVATORY-AUTH')
        if not authenticate_user(token):
            return make_response(jsonify({'error': 'Not authorized'}), 401)

        # Remove token on logout
        del tokens[token]
        return jsonify({'message': 'Logout successful'})

# Add resources to the API
api.add_resource(MovieRecommendation, '/ntuaflix_api/movie')
api.add_resource(UserLogin, '/ntuaflix_api/login')
api.add_resource(UserLogout, '/ntuaflix_api/logout')

if __name__ == '__main__':
    app.run(debug=True, ssl_context='adhoc', port=9876)
This example includes three endpoints:

/ntuaflix_api/movie: GET endpoint for movie recommendations with optional format query parameter.
/ntuaflix_api/login: POST endpoint for user login.
/ntuaflix_api/logout: POST endpoint for user logout.
Please note that this code uses in-memory storage for users and tokens, which is not suitable for a production environment. You should implement proper user authentication and token management based on your requirements.

from flask import Flask, request, jsonify, make_response
from flask_sqlalchemy import SQLAlchemy
from flask_restful import Api, Resource

app = Flask(__name__)
app.config['SQLALCHEMY_DATABASE_URI'] = 'mysql://your_username:your_password@localhost/your_database_name'
app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False

db = SQLAlchemy(app)
api = Api(app)

# Movie model
class Movie(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    title = db.Column(db.String(255), nullable=False)
    genre = db.Column(db.String(255), nullable=False)

# User model
class User(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    username = db.Column(db.String(255), unique=True, nullable=False)
    password = db.Column(db.String(255), nullable=False)

# Token model (for demonstration purposes)
class Token(db.Model):
    id = db.Column(db.Integer, primary_key=True)
    token = db.Column(db.String(255), unique=True, nullable=False)
    user_id = db.Column(db.Integer, db.ForeignKey('user.id'), nullable=False)

# Resource for movie recommendation
class MovieRecommendation(Resource):
    def get(self):
        # Check if user is authenticated
        token = request.headers.get('X-OBSERVATORY-AUTH')
        user = User.query.join(Token).filter(Token.token == token).first()

        if not user:
            return make_response(jsonify({'error': 'Not authorized'}), 401)

        # Get format parameter from the query string, default to 'json'
        data_format = request.args.get('format', 'json')

        if data_format == 'json':
            movies = Movie.query.all()
            return jsonify([{'id': movie.id, 'title': movie.title, 'genre': movie.genre} for movie in movies])
        elif data_format == 'csv':
            # Generate CSV response
            csv_data = "id,title,genre\n"
            movies = Movie.query.all()
            for movie in movies:
                csv_data += f"{movie.id},{movie.title},{movie.genre}\n"

            response = make_response(csv_data)
            response.headers['Content-Type'] = 'text/csv; charset=utf-8'
            return response
        else:
            return make_response(jsonify({'error': 'Invalid format parameter'}), 400)

# Resource for user login
class UserLogin(Resource):
    def post(self):
        username = request.form.get('username')
        password = request.form.get('password')

        user = User.query.filter_by(username=username, password=password).first()

        if user:
            # Generate and store token
            token = f"{username}_token"
            new_token = Token(token=token, user_id=user.id)
            db.session.add(new_token)
            db.session.commit()

            return jsonify({'token': token})
        else:
            return make_response(jsonify({'error': 'Bad request'}), 400)

# Resource for user logout
class UserLogout(Resource):
    def post(self):
        # Check if user is authenticated
        token = request.headers.get('X-OBSERVATORY-AUTH')
        user = User.query.join(Token).filter(Token.token == token).first()

        if not user:
            return make_response(jsonify({'error': 'Not authorized'}), 401)

        # Remove token on logout
        Token.query.filter_by(token=token).delete()
        db.session.commit()

        return jsonify({'message': 'Logout successful'})

# Add resources to the API
api.add_resource(MovieRecommendation, '/ntuaflix_api/movie')
api.add_resource(UserLogin, '/ntuaflix_api/login')
api.add_resource(UserLogout, '/ntuaflix_api/logout')

if __name__ == '__main__':
    db.create_all()
    app.run(debug=True, ssl_context='adhoc', port=9876)
Make sure to replace your_username, your_password, and your_database_name with your actual MySQL database credentials and database name.
